182  void recordCg()
183  {
184  // input vector, needs to be dynamic size
185  Eigen::Matrix<CG_SCALAR, Eigen::Dynamic, 1> x(inputDim_);
186  // init to rand to avoid floating point problems in user's code
187  x.setRandom();
188 
189  // declare x as independent
190  CppAD::Independent(x);
191 
192  // output vector, needs to be dynamic size
193  Eigen::Matrix<CG_SCALAR, Eigen::Dynamic, 1> y(outputDim_);
194 
195  y = cgStdFun_(x);
196 
197  // store operation sequence in f: x -> y and stop recording
198  CppAD::ADFun<CG_VALUE_TYPE> fCodeGen(x, y);
199 
200  fCodeGen.optimize();
201 
202  cgCppadFun_ = fCodeGen;
203  }
