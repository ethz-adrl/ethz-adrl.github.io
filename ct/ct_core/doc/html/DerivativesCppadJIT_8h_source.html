359  void compileJIT(const DerivativesCppadSettings& settings,
360  const std::string& libName = "unnamedLib",
361  bool verbose = false)
362  {
363  if (compiled_)
364  return;
365 
366  struct timespec ts;
367  clock_gettime(CLOCK_MONOTONIC, &ts);
368 
369  // assigning a unique identifier to the library in order to avoid race conditions in JIT
370  std::string uniqueID =
371  std::to_string(std::hash<std::thread::id>()(std::this_thread::get_id())) + "_" + std::to_string(ts.tv_nsec);
372 
373  libName_ = libName + uniqueID;
374 
375  CppAD::cg::ModelCSourceGen<double> cgen(cgCppadFun_, "DerivativesCppad" + libName_);
376 
377  cgen.setMultiThreading(settings.multiThreading_);
378  cgen.setCreateForwardZero(settings.createForwardZero_);
379  cgen.setCreateForwardOne(settings.createForwardOne_);
380  cgen.setCreateReverseOne(settings.createReverseOne_);
381  cgen.setCreateReverseTwo(settings.createReverseTwo_);
382  cgen.setCreateJacobian(settings.createJacobian_);
383  cgen.setCreateSparseJacobian(settings.createSparseJacobian_);
384  cgen.setCreateHessian(settings.createHessian_);
385  cgen.setCreateSparseHessian(settings.createSparseHessian_);
386  cgen.setMaxAssignmentsPerFunc(settings.maxAssignements_);
387 
388  CppAD::cg::ModelLibraryCSourceGen<double> libcgen(cgen);
389  std::string tempDir = "cppad_temp" + uniqueID;
390  if (verbose)
391  {
392  std::cout << "DerivativesCppadJIT: starting to compile " << libName_ << " library ..." << std::endl;
393  std::cout << "DerivativesCppadJIT: in temporary directory " << tempDir << std::endl;
394  }
395 
396  // compile source code
397  CppAD::cg::DynamicModelLibraryProcessor<double> p(libcgen, libName_);
398  if (settings.compiler_ == DerivativesCppadSettings::GCC)
399  {
400  CppAD::cg::GccCompiler<double> compiler;
401  compiler.setTemporaryFolder(tempDir);
402  dynamicLib_ = std::shared_ptr<CppAD::cg::DynamicLib<double>>(p.createDynamicLibrary(compiler));
403  }
404  else if (settings.compiler_ == DerivativesCppadSettings::CLANG)
405  {
406  CppAD::cg::ClangCompiler<double> compiler;
407  compiler.setTemporaryFolder(tempDir);
408  dynamicLib_ = std::shared_ptr<CppAD::cg::DynamicLib<double>>(p.createDynamicLibrary(compiler));
409  }
410 
411  if (settings.generateSourceCode_)
412  {
413  CppAD::cg::SaveFilesModelLibraryProcessor<double> p2(libcgen);
414  p2.saveSources();
415  }
416 
417  model_ = std::shared_ptr<CppAD::cg::GenericModel<double>>(dynamicLib_->model("DerivativesCppad" + libName_));
418 
419  compiled_ = true;
420 
421  if (verbose)
422  std::cout << "DerivativesCppadJIT: successfully compiled " << libName_ << std::endl;
423 
424  if (model_->isJacobianSparsityAvailable())
425  {
426  model_->JacobianSparsity(sparsityRowsJacobian_, sparsityColsJacobian_);
427  sparsityRowsJacobianEigen_.resize(sparsityRowsJacobian_.size()); //rowsTmp.resize(rowsVec.size());
428  sparsityColsJacobianEigen_.resize(sparsityColsJacobian_.size()); //colsTmp.resize(rowsVec.size());
429 
430  Eigen::Matrix<size_t, Eigen::Dynamic, 1> rowsSizeT;
431  rowsSizeT.resize(sparsityRowsJacobian_.size());
432  Eigen::Matrix<size_t, Eigen::Dynamic, 1> colsSizeT;
433  colsSizeT.resize(sparsityColsJacobian_.size());
434 
435  rowsSizeT = Eigen::Map<Eigen::Matrix<size_t, Eigen::Dynamic, 1>>(
436  sparsityRowsJacobian_.data(), sparsityRowsJacobian_.size(), 1);
437  colsSizeT = Eigen::Map<Eigen::Matrix<size_t, Eigen::Dynamic, 1>>(
438  sparsityColsJacobian_.data(), sparsityColsJacobian_.size(), 1);
439 
440  sparsityRowsJacobianEigen_ = rowsSizeT.cast<int>();
441  sparsityColsJacobianEigen_ = colsSizeT.cast<int>();
442  }
443 
444  if (model_->isHessianSparsityAvailable())
445  {
446  model_->HessianSparsity(sparsityRowsHessian_, sparsityColsHessian_);
447  sparsityRowsHessianEigen_.resize(sparsityRowsHessian_.size()); //rowsTmp.resize(rowsVec.size());
448  sparsityColsHessianEigen_.resize(sparsityColsHessian_.size()); //colsTmp.resize(rowsVec.size());
449 
450  Eigen::Matrix<size_t, Eigen::Dynamic, 1> rowsSizeT;
451  rowsSizeT.resize(sparsityRowsHessian_.size());
452  Eigen::Matrix<size_t, Eigen::Dynamic, 1> colsSizeT;
453  colsSizeT.resize(sparsityColsHessian_.size());
454 
455  rowsSizeT = Eigen::Map<Eigen::Matrix<size_t, Eigen::Dynamic, 1>>(
456  sparsityRowsHessian_.data(), sparsityRowsHessian_.size(), 1);
457  colsSizeT = Eigen::Map<Eigen::Matrix<size_t, Eigen::Dynamic, 1>>(
458  sparsityColsHessian_.data(), sparsityColsHessian_.size(), 1);
459 
460  sparsityRowsHessianEigen_ = rowsSizeT.cast<int>();
461  sparsityColsHessianEigen_ = colsSizeT.cast<int>();

470  {
471  // input vector, needs to be dynamic size
472  Eigen::Matrix<CG_SCALAR, Eigen::Dynamic, 1> x(inputDim_);
473 
474  // declare x as independent
475  CppAD::Independent(x);

479 
480  y = cgStdFun_(x);
481 
482  // store operation sequence in f: x -> y and stop recording
483  CppAD::ADFun<CG_VALUE_TYPE> fCodeGen(x, y);
484 
485  fCodeGen.optimize();
486 
487  cgCppadFun_ = fCodeGen;
488  }
