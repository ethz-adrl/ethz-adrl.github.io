141  void compileJIT(const std::string& libName = "DynamicsLinearizerADCG", bool verbose = false)
142  {
143  if (compiled_)
144  return;
145 
146  struct timespec ts;
147  clock_gettime(CLOCK_MONOTONIC, &ts);
148 
149  // assigning a unique identifier to the library in order to avoid race conditions in JIT
150  std::string uniqueID =
151  std::to_string(std::hash<std::thread::id>()(std::this_thread::get_id())) + "_" + std::to_string(ts.tv_nsec);
152 
153  jitLibName_ = libName + uniqueID;
154 
155  CppAD::cg::ModelCSourceGen<OUT_SCALAR> cgen(this->f_, "DynamicsLinearizerADCG" + jitLibName_);
156  cgen.setCreateJacobian(true);
157  CppAD::cg::ModelLibraryCSourceGen<OUT_SCALAR> libcgen(cgen);
158  std::string tempDir = "cppad_temp" + uniqueID;
159  if (verbose)
160  {
161  std::cout << "Starting to compile " << jitLibName_ << " library ..." << std::endl;
162  std::cout << "In temporary directory " << tempDir << std::endl;
163  }
164 
165  // compile source code
166  CppAD::cg::DynamicModelLibraryProcessor<OUT_SCALAR> p(libcgen, jitLibName_);
167  compiler_.setTemporaryFolder(tempDir);
168  dynamicLib_ = std::shared_ptr<CppAD::cg::DynamicLib<OUT_SCALAR>>(p.createDynamicLibrary(compiler_));
169 
170  model_ = std::shared_ptr<CppAD::cg::GenericModel<OUT_SCALAR>>(
171  dynamicLib_->model("DynamicsLinearizerADCG" + jitLibName_));
172 
173  compiled_ = true;
174 
175  if (verbose)
176  std::cout << "Successfully compiled " << jitLibName_ << std::endl;
177  }

224  void computeJacobian(const state_vector_t& x, const control_vector_t& u)
225  {
226  // copy to dynamic type due to requirements by cppad
227  Eigen::Matrix<OUT_SCALAR, Eigen::Dynamic, 1> input(STATE_DIM + CONTROL_DIM);
228  input << x, u;
229 
230  Eigen::Matrix<OUT_SCALAR, Eigen::Dynamic, 1> jac(Base::FullJac_entries);
231 
232  jac = model_->Jacobian(input);
233 
234  Eigen::Map<Eigen::Matrix<OUT_SCALAR, STATE_DIM + CONTROL_DIM, STATE_DIM>> out(jac.data());
235 
236  dFdx_ = out.template topRows<STATE_DIM>().transpose();
237  dFdu_ = out.template bottomRows<CONTROL_DIM>().transpose();
238 
239  x_at_cache_ = x;
240  u_at_cache_ = u;
241  }
