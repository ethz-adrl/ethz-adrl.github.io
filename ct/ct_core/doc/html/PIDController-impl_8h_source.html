53 template <typename SCALAR>
54 SCALAR PIDController<SCALAR>::computeControl(const SCALAR& state, const Time& t)
55 {
56  SCALAR error = setpoint_.stateDesired_ - state;
57 
58  // ** P-part **
59  // ============
60  SCALAR P = parameters_.k_p * error;
61 
62  // ** I-part **
63  // ============
64  computeI(error);
65 
66  // ** D-Part **
67  // ============
68  SCALAR D = parameters_.k_d * (setpoint_.stateDerivativeDesired_ - (state - statePrevious_) / parameters_.dt);
69 
70  // ** Controller Output **
71  // =======================
72  SCALAR u = (P + I_ + D);
73  saturateControl(u);
74 
75  statePrevious_ = state;
76 
77  return u;
78 }
