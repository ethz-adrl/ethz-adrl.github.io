144 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>
145 void StateFeedbackController<STATE_DIM, CONTROL_DIM, SCALAR>::extractControlTrajectory(
146  const StateTrajectory<STATE_DIM, SCALAR>& x_traj,
147  ControlTrajectory<CONTROL_DIM, SCALAR>& u_traj)
148 {
149  u_traj.clear();
150 
151  for (size_t i = 0; i < x_traj.size() - 1; i++)
152  {
153  u_traj.push_back(uff_[i] + K_[i] * (x_traj[i] - x_ref_[i]), x_traj.getTimeFromIndex(i), true);
154  }
155 }
