152 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>

154  Eigen::VectorXi& iRow_vec,
155  Eigen::VectorXi& jCol_vec)
156 {
157  // make sure the sparsity pattern is correct and consists only of ones and zeros
158  assert(diag_sparsity.maxCoeff() <= 1);
159  assert(diag_sparsity.minCoeff() >= 0);
160 
161  const int num_elements = diag_sparsity.sum();
162 
163  iRow_vec.resize(num_elements);
164  jCol_vec.resize(num_elements);
165 
166  size_t count = 0;
167 
168  for (int i = 0; i < diag_sparsity.rows(); ++i)
169  {
170  if (diag_sparsity(i) == 1)
171  {
172  iRow_vec(count) = i;
173  jCol_vec(count) = i;
174  count++;
175  }
176  }
177 }
