392 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>
393 bool ConstraintContainerAD<STATE_DIM, CONTROL_DIM, SCALAR>::initializeIntermediate()
394 {
395  Eigen::VectorXi sparsityRows;
396  Eigen::VectorXi sparsityCols;
397 
398  if (getIntermediateConstraintsCount() > 0)
399  {
400  ct::core::DerivativesCppadSettings settings;
401  settings.createForwardZero_ = true;
402  settings.createJacobian_ = true;
403  settings.createSparseJacobian_ = true;
404 
405  intermediateCodegen_->compileJIT(settings, "intermediateConstraints");
406  intermediateCodegen_->getSparsityPatternJacobian(sparsityRows, sparsityCols);
407 
408  std::cout << "sparsityPattern Intermediate: " << std::endl
409  << intermediateCodegen_->getSparsityPatternJacobian() << std::endl;
410  assert(sparsityRows.rows() == sparsityRows.rows());
411 
412  int nonZerosState = (sparsityCols.array() < STATE_DIM).count();
413  int nonZerosInput = (sparsityCols.array() >= STATE_DIM).count();
414 
415  sparsityStateIntermediateRows_.resize(nonZerosState);
416  sparsityStateIntermediateCols_.resize(nonZerosState);
417  sparsityInputIntermediateRows_.resize(nonZerosInput);
418  sparsityInputIntermediateCols_.resize(nonZerosInput);
419 
420  size_t count = 0;
421 
422  this->lowerBoundsIntermediate_.resize(getIntermediateConstraintsCount());
423  this->upperBoundsIntermediate_.resize(getIntermediateConstraintsCount());
424 
425  for (auto constraint : constraintsIntermediate_)
426  {
427  size_t constraintSize = constraint->getConstraintSize();
428  this->lowerBoundsIntermediate_.segment(count, constraintSize) = constraint->getLowerBound();
429  this->upperBoundsIntermediate_.segment(count, constraintSize) = constraint->getUpperBound();
430  count += constraintSize;
431  }
432 
433  size_t stateIndex = 0;
434  size_t inputIndex = 0;
435 
436  for (int i = 0; i < sparsityRows.rows(); ++i)
437  {
438  if (sparsityCols(i) < static_cast<int>(STATE_DIM))
439  {
440  sparsityStateIntermediateRows_(stateIndex) = sparsityRows(i);
441  sparsityStateIntermediateCols_(stateIndex) = sparsityCols(i);
442  stateIndex++;
443  }
444  else
445  {
446  sparsityInputIntermediateRows_(inputIndex) = sparsityRows(i);
447  sparsityInputIntermediateCols_(inputIndex) = sparsityCols(i) - STATE_DIM;
448  inputIndex++;
449  }
450  }
451  }
452 
453  return true;
454 }
