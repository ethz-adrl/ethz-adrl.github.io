133 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>
134 void CostFunctionAD<STATE_DIM, CONTROL_DIM, SCALAR>::loadFromConfigFile(const std::string& filename, bool verbose)
135 {
136  this->intermediateCostAnalytical_.clear();
137  this->finalCostAnalytical_.clear();
138 
139  boost::property_tree::ptree pt;
140  boost::property_tree::read_info(filename, pt);
141  int i = 0;
142  std::string currentTerm;
143  do
144  {
145  currentTerm = "term" + std::to_string(i);
146  std::string termKind = pt.get<std::string>(currentTerm + ".kind");
147  boost::algorithm::to_lower(termKind);
148  int currentTermType = pt.get<int>(currentTerm + ".type");
149  std::string termName;
150  try
151  {
152  termName = pt.get<std::string>(currentTerm + ".name");
153  } catch (boost::property_tree::ptree_bad_path err)
154  {
155  termName = "Unnamed";
156  if (verbose)
157  {
158  std::cout << "Name field for " + currentTerm + " does not exist" << std::endl;
159  }
160  }
161 
162  std::shared_ptr<TermBase<STATE_DIM, CONTROL_DIM, SCALAR, CGScalar>> term;
163 
164  CT_LOADABLE_TERMS(SCALAR, CGScalar);
165 
166  if (!term)
167  {
168  throw std::runtime_error("Term type \"" + termKind + "\" not supported");
169  }
170  else
171  {
172  if (term)
173  addADTerm(filename, currentTerm, currentTermType, term, this, verbose);
174  else
175  throw std::runtime_error("Term type \"" + termKind + "\" loaded but unsupported.");
176  }
177  currentTerm = "term" + std::to_string(++i);
178  } while (pt.find(currentTerm) != pt.not_found());
179 }

183 CostFunctionAD<STATE_DIM, CONTROL_DIM, SCALAR>::evaluateIntermediateCg(
184  const Eigen::Matrix<CGScalar, STATE_DIM + CONTROL_DIM + 1, 1>& stateInputTime)
185 {
186  CGScalar y = CGScalar(0.0);
187 
188  for (auto it : intermediateTerms_)
189  y += it->evaluateCppadCg(stateInputTime.segment(0, STATE_DIM), stateInputTime.segment(STATE_DIM, CONTROL_DIM),

191 
192  Eigen::Matrix<CGScalar, 1, 1> out;
193  out << y;
194  return out;
