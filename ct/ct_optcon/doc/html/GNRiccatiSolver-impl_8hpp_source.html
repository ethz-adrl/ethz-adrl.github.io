64 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>
65 void GNRiccatiSolver<STATE_DIM, CONTROL_DIM, SCALAR>::extractLQSolution()
66 {
67  LQOCProblem_t& p = *this->lqocProblem_;
68 
69  this->x_sol_[0] = p.x_[0];
70 
71  for (int k = 0; k < this->lqocProblem_->getNumberOfStages(); k++)
72  {
74  this->u_sol_[k] = lv_[k] + this->L_[k] * this->x_sol_[k];
75 
77  this->x_sol_[k + 1] = p.A_[k] * this->x_sol_[k] + p.B_[k] * this->u_sol_[k] + p.b_[k];
78  }
79 }

147 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>
148 void GNRiccatiSolver<STATE_DIM, CONTROL_DIM, SCALAR>::computeCostToGo(size_t k)
149 {
150  LQOCProblem_t& p = *this->lqocProblem_;
151 
152  S_[k] = p.Q_[k];
153  S_[k].noalias() += p.A_[k].transpose() * S_[k + 1] * p.A_[k];
154  S_[k].noalias() -= this->L_[k].transpose() * Hi_[k] * this->L_[k];
155 
156  S_[k] = 0.5 * (S_[k] + S_[k].transpose()).eval();
157 
158  sv_[k] = p.qv_[k];
159  sv_[k].noalias() += p.A_[k].transpose() * sv_[k + 1];
160  sv_[k].noalias() += p.A_[k].transpose() * S_[k + 1] * p.b_[k];
161  sv_[k].noalias() += this->L_[k].transpose() * Hi_[k] * lv_[k];
162  sv_[k].noalias() += this->L_[k].transpose() * gv_[k];
163  sv_[k].noalias() += G_[k].transpose() * lv_[k];
164 }

167 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR>
168 void GNRiccatiSolver<STATE_DIM, CONTROL_DIM, SCALAR>::designController(size_t k)
169 {
170  LQOCProblem_t& p = *this->lqocProblem_;
171 
172  gv_[k] = p.rv_[k];
173  gv_[k].noalias() += p.B_[k].transpose() * sv_[k + 1];
174  gv_[k].noalias() += p.B_[k].transpose() * S_[k + 1].template selfadjointView<Eigen::Lower>() * p.b_[k];
175 
176  G_[k] = p.P_[k];
177  //G_[k].noalias() += B_[k].transpose() * S_[k+1] * A_[k];
178  G_[k].noalias() += p.B_[k].transpose() * S_[k + 1].template selfadjointView<Eigen::Lower>() * p.A_[k];
179 
180  H_[k] = p.R_[k];
181  //H_[k].noalias() += B_[k].transpose() * S_[k+1] * B_[k];
182  H_[k].noalias() += p.B_[k].transpose() * S_[k + 1].template selfadjointView<Eigen::Lower>() * p.B_[k];
183 
184  if (settings_.fixedHessianCorrection)
185  {
186  if (settings_.epsilon > 1e-10)
187  Hi_[k] = H_[k] + settings_.epsilon * ControlMatrix::Identity();
188  else
189  Hi_[k] = H_[k];
190 
191  if (settings_.recordSmallestEigenvalue)
192  {
193  // compute eigenvalues with eigenvectors enabled
194  eigenvalueSolver_.compute(Hi_[k], Eigen::ComputeEigenvectors);
195  const ControlMatrix& V = eigenvalueSolver_.eigenvectors().real();
196  const ControlVector& lambda = eigenvalueSolver_.eigenvalues();
197 
198  smallestEigenvalue_ = std::min(smallestEigenvalue_, lambda.minCoeff());
199 
200  // Corrected Eigenvalue Matrix
201  ControlMatrix D = ControlMatrix::Zero();
202  // make D positive semi-definite (as described in IV. B.)
203  D.diagonal() = lambda.cwiseMax(settings_.epsilon);
204 
205  // reconstruct H
206  ControlMatrix Hi_regular = V * D * V.transpose();
207 
208  // invert D
209  ControlMatrix D_inverse = ControlMatrix::Zero();
210  // eigenvalue-wise inversion
211  D_inverse.diagonal() = -1.0 * D.diagonal().cwiseInverse();
212  ControlMatrix Hi_inverse_regular = V * D_inverse * V.transpose();
213 
214  if (!Hi_inverse_[k].isApprox(Hi_inverse_regular, 1e-4))
215  {
216  std::cout << "warning, inverses not identical at " << k << std::endl;
217  std::cout << "Hi_inverse_fixed - Hi_inverse_regular: " << std::endl
218  << Hi_inverse_[k] - Hi_inverse_regular << std::endl
219  << std::endl;
220  }
221  }
222 
223  Hi_inverse_[k] = -Hi_[k].template selfadjointView<Eigen::Lower>().llt().solve(ControlMatrix::Identity());
224 
225  // calculate FB gain update
226  this->L_[k].noalias() = Hi_inverse_[k].template selfadjointView<Eigen::Lower>() * G_[k];
227 
228  // calculate FF update
229  lv_[k].noalias() = Hi_inverse_[k].template selfadjointView<Eigen::Lower>() * gv_[k];
230  }
231  else
232  {
233  // compute eigenvalues with eigenvectors enabled
234  eigenvalueSolver_.compute(H_[k], Eigen::ComputeEigenvectors);
235  const ControlMatrix& V = eigenvalueSolver_.eigenvectors().real();
236  const ControlVector& lambda = eigenvalueSolver_.eigenvalues();
237 
238  if (settings_.recordSmallestEigenvalue)
239  {
240  smallestEigenvalue_ = std::min(smallestEigenvalue_, lambda.minCoeff());
241  }
242 
243  // Corrected Eigenvalue Matrix
244  ControlMatrix D = ControlMatrix::Zero();
245  // make D positive semi-definite (as described in IV. B.)
246  D.diagonal() = lambda.cwiseMax(settings_.epsilon);
247 
248  // reconstruct H
249  Hi_[k].noalias() = V * D * V.transpose();
250 
251  // invert D
252  ControlMatrix D_inverse = ControlMatrix::Zero();
253  // eigenvalue-wise inversion
254  D_inverse.diagonal() = -1.0 * D.diagonal().cwiseInverse();
255  Hi_inverse_[k].noalias() = V * D_inverse * V.transpose();
256 
257  // calculate FB gain update
258  this->L_[k].noalias() = Hi_inverse_[k] * G_[k];
259 
260  // calculate FF update
261  lv_[k].noalias() = Hi_inverse_[k] * gv_[k];
262  }
263 }
