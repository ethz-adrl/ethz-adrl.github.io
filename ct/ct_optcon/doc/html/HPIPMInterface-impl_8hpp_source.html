168 template <int STATE_DIM, int CONTROL_DIM>
169 void HPIPMInterface<STATE_DIM, CONTROL_DIM>::designFeedback()
170 {
171  LQOCProblem<STATE_DIM, CONTROL_DIM>& p = *this->lqocProblem_;
172  this->L_.resize(p.getNumberOfStages());
173 
174  // for stage 0, HPIPM does not provide feedback, so we have to construct it
175 
176  // step 1: reconstruct H[0]
177  Eigen::Matrix<double, control_dim, control_dim> Lr;
178  ::d_cvt_strmat2mat(Lr.rows(), Lr.cols(), &workspace_.L[0], 0, 0, Lr.data(), Lr.rows());
179  Eigen::Matrix<double, control_dim, control_dim> H;
180  H = Lr.template triangularView<Eigen::Lower>() * Lr.transpose(); // Lr is cholesky of H
181 
182  // step2: reconstruct S[1]
183  Eigen::Matrix<double, state_dim, state_dim> Lq;
184  ::d_cvt_strmat2mat(Lq.rows(), Lq.cols(), &workspace_.L[1], control_dim, control_dim, Lq.data(), Lq.rows());
185  Eigen::Matrix<double, state_dim, state_dim> S;
186  S = Lq.template triangularView<Eigen::Lower>() * Lq.transpose(); // Lq is cholesky of S
187 
188  // step3: compute G[0]
189  Eigen::Matrix<double, control_dim, state_dim> G;
190  G = p.P_[0];
191  G.noalias() += p.B_[0].transpose() * S * p.A_[0];
192 
193  // step4: compute K[0]
194  this->L_[0] = (-H.inverse() * G); // \todo use Lr here instead of H!
195 
196  // for all other steps we can just read Ls
197  Eigen::Matrix<double, state_dim, control_dim> Ls;
198  for (int i = 1; i < this->lqocProblem_->getNumberOfStages(); i++)
199  {
200  ::d_cvt_strmat2mat(Lr.rows(), Lr.cols(), &workspace_.L[i], 0, 0, Lr.data(), Lr.rows());
201  ::d_cvt_strmat2mat(Ls.rows(), Ls.cols(), &workspace_.L[i], Lr.rows(), 0, Ls.data(), Ls.rows());
202  this->L_[i] = (-Ls * Lr.partialPivLu().inverse()).transpose();
203  }
204 }

295 void HPIPMInterface<STATE_DIM, CONTROL_DIM>::setProblemImpl(
296  std::shared_ptr<LQOCProblem<STATE_DIM, CONTROL_DIM>> lqocProblem)
297 {
298  // check if the number of stages N changed and adapt problem dimensions
299  bool nStagesChanged = changeNumberOfStages(lqocProblem->getNumberOfStages());
300 
301 
302  // WARNING: the allocation should in practice not have to happen during the loop.
303  // If possible, prefer receding horizon MPC problems.
304  // If the number of stages has changed, however, the problem needs to be re-built:
305  if (nStagesChanged)
306  {
307  // update constraint configuration in case the horizon length has changed.
308  if (lqocProblem->isBoxConstrained())
309  configureBoxConstraints(lqocProblem);
310 
311  if (lqocProblem->isGeneralConstrained())
312  configureGeneralConstraints(lqocProblem);
313  }
314 
315  // setup unconstrained part of problem
316  setupCostAndDynamics(lqocProblem->x_, lqocProblem->u_, lqocProblem->A_, lqocProblem->B_, lqocProblem->b_,
317  lqocProblem->P_, lqocProblem->qv_, lqocProblem->Q_, lqocProblem->rv_, lqocProblem->R_);
318 
319 
320  if (nStagesChanged)
321  {
322  initializeAndAllocate();
323  }
324 }

328 void HPIPMInterface<STATE_DIM, CONTROL_DIM>::configureBoxConstraints(
329  std::shared_ptr<LQOCProblem<STATE_DIM, CONTROL_DIM>> lqocProblem)
330 {
331  // stages 1 to N
332  for (int i = 0; i < N_ + 1; i++)
333  {
334  nb_[i] = lqocProblem->nb_[i];
335 
336  // set pointers to box constraint boundaries and sparsity pattern
337  hd_lb_[i] = lqocProblem->ux_lb_[i].data();
338  hd_ub_[i] = lqocProblem->ux_ub_[i].data();
339  hidxb_[i] = lqocProblem->ux_I_[i].data();

342  if (i == 0)
343  {

346  {
347  if (lqocProblem->ux_I_[i](j) < CONTROL_DIM)
348  nb_[i]++; // adapt number of constraints such that only controls are listed as decision vars
349  else
350  break;
351  }
352  }
353 
354  // TODO clarify with Gianluca if we need to reset the lagrange multiplier
355  // before warmstarting (potentially wrong warmstart for the lambdas)
356 
357  // direct pointers of lagrange mult to corresponding containers
358  lam_lb_[i] = cont_lam_lb_[i].data();
359  lam_ub_[i] = cont_lam_ub_[i].data();
360  }

364 template <int STATE_DIM, int CONTROL_DIM>
365 void HPIPMInterface<STATE_DIM, CONTROL_DIM>::configureGeneralConstraints(
366  std::shared_ptr<LQOCProblem<STATE_DIM, CONTROL_DIM>> lqocProblem)
367 {
368  // HPIPM-specific correction for first-stage general constraint bounds
369  hd_lg_0_Eigen_ = lqocProblem->d_lb_[0] - lqocProblem->C_[0] * lqocProblem->x_[0];
370  hd_ug_0_Eigen_ = lqocProblem->d_ub_[0] - lqocProblem->C_[0] * lqocProblem->x_[0];
371 
372  for (int i = 0; i < N_ + 1; i++) // (includes the terminal stage)
373  {
374  // check dimensions
375  assert(lqocProblem->d_lb_[i].rows() == lqocProblem->d_ub_[i].rows());
376  assert(lqocProblem->d_lb_[i].rows() == lqocProblem->C_[i].rows());
377  assert(lqocProblem->d_lb_[i].rows() == lqocProblem->D_[i].rows());
378  assert(lqocProblem->C_[i].cols() == STATE_DIM);
379  assert(lqocProblem->D_[i].cols() == CONTROL_DIM);
380 
381  // get the number of constraints
382  ng_[i] = lqocProblem->ng_[i];
383 
384  // set pointers to hpipm-style box constraint boundaries and sparsity pattern
385  if (i == 0)
386  {
387  hd_lg_[i] = hd_lg_0_Eigen_.data();
388  hd_ug_[i] = hd_ug_0_Eigen_.data();
389  }
390  else
391  {
392  hd_lg_[i] = lqocProblem->d_lb_[i].data();
393  hd_ug_[i] = lqocProblem->d_ub_[i].data();
394  }
395  hC_[i] = lqocProblem->C_[i].data();
396  hD_[i] = lqocProblem->D_[i].data();
397 
398  // TODO clarify with Gianluca if we need to reset the lagrange multiplier
399  // before warmstarting (potentially wrong warmstart for the lambdas)
400 
401  // direct pointers of lagrange mult to corresponding containers
402  cont_lam_lg_[i].resize(ng_[i]); // todo avoid dynamic allocation (e.g. by defining a max. constraint dim)
403  cont_lam_ug_[i].resize(ng_[i]); // todo avoid dynamic allocation (e.g. by defining a max. constraint dim)
404  lam_lg_[i] = cont_lam_lg_[i].data();
405  lam_ug_[i] = cont_lam_ug_[i].data();

464 {

467 
468  N_ = N;
469 
470  nx_.resize(N_ + 1, STATE_DIM); // initialize number of states per stage
471  nu_.resize(N_ + 1, CONTROL_DIM); // initialize number of control inputs per stage
472  nb_.resize(N_ + 1, nb_.back()); // initialize number of box constraints per stage
473  ng_.resize(N_ + 1, ng_.back()); // initialize number of general constraints per stage
474 
475  // resize the containers for the affine system dynamics approximation
476  hA_.resize(N_);
477  hB_.resize(N_);
478  hb_.resize(N_);
479 
480  // resize the containers for the LQ cost approximation
481  hQ_.resize(N_ + 1);
482  hS_.resize(N_ + 1);
483  hR_.resize(N_ + 1);
484  hq_.resize(N_ + 1);
485  hr_.resize(N_ + 1);
486 
487  hd_lb_.resize(N_ + 1);
488  hd_ub_.resize(N_ + 1);
489  hidxb_.resize(N_ + 1);
490  hd_lg_.resize(N_ + 1);
491  hd_ug_.resize(N_ + 1);
492  hC_.resize(N_ + 1);
493  hD_.resize(N_ + 1);
494 
495  u_.resize(N_ + 1);
496  x_.resize(N_ + 1);
497  pi_.resize(N_);
498  hpi_.resize(N_);
499  this->x_sol_.resize(N_ + 1);
500  this->u_sol_.resize(N_);
501 
502  lam_lb_.resize(N_ + 1);
503  lam_ub_.resize(N_ + 1);
504  lam_lg_.resize(N_ + 1);
505  lam_ug_.resize(N_ + 1);
506  cont_lam_lb_.resize(N_ + 1);
507  cont_lam_ub_.resize(N_ + 1);
508  cont_lam_lg_.resize(N_ + 1);
509  cont_lam_ug_.resize(N_ + 1);
510 
511  for (int i = 0; i < N_; i++)
512  {
513  // first state and last input are not optimized
514  x_[i + 1] = this->x_sol_[i + 1].data();
515  u_[i] = this->u_sol_[i].data();
516  }
517  for (int i = 0; i < N_; i++)
518  {
519  pi_[i] = hpi_[i].data();
520  }
521 
522  hS_[N_] = nullptr;
523  hR_[N_] = nullptr;
524  hr_[N_] = nullptr;
525
