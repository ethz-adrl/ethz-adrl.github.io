105  d_ub_[i].setZero();

111 
112  hasBoxConstraints_ = false;
113  hasGenConstraints_ = false;
114 }

117 template <int STATE_DIM, int CONTROL_DIM, typename SCALAR>
118 void LQOCProblem<STATE_DIM, CONTROL_DIM, SCALAR>::setIntermediateBoxConstraint(const int index,
119  const int nConstr,
120  const constr_vec_t& ux_lb,
121  const constr_vec_t& ux_ub,
122  const VectorXi& sp)
123 {
124  if ((ux_lb.rows() != ux_ub.rows()) | (ux_lb.size() != nConstr) | (sp.rows() != nConstr) |
125  (sp(sp.rows() - 1) > (STATE_DIM + CONTROL_DIM - 1)))
126  {
127  std::cout << "n.o. constraints : " << nConstr << std::endl;
128  std::cout << "ux_lb : " << ux_lb.transpose() << std::endl;
129  std::cout << "ux_ub : " << ux_ub.transpose() << std::endl;
130  std::cout << "sparsity : " << sp.transpose() << std::endl;
131  throw(std::runtime_error("LQOCProblem setIntermediateBoxConstraint: error in constraint config"));
132  }
133 
134  if (index >= K_)
135  throw(std::runtime_error("LQOCProblem cannot set an intermediate Box constraint at time >= K_"));
136 
137  nb_[index] = nConstr;
138  ux_lb_[index].template topRows(nConstr) = ux_lb;
139  ux_ub_[index].template topRows(nConstr) = ux_ub;
140  ux_I_[index].template topRows(nConstr) = sp;
141 
142  hasBoxConstraints_ = true;
143 }

198 template <int STATE_DIM, int CONTROL_DIM, typename SCALAR>
199 void LQOCProblem<STATE_DIM, CONTROL_DIM, SCALAR>::setFromTimeInvariantLinearQuadraticProblem(
200  ct::core::StateVector<STATE_DIM, SCALAR>& x0,
201  ct::core::ControlVector<CONTROL_DIM, SCALAR>& u0,
202  ct::core::DiscreteLinearSystem<STATE_DIM, CONTROL_DIM, SCALAR>& linearSystem,
203  ct::optcon::CostFunctionQuadratic<STATE_DIM, CONTROL_DIM, SCALAR>& costFunction,
204  ct::core::StateVector<STATE_DIM, SCALAR>& offset,
205  double dt)
206 {
207  setZero();
208 
209  x_ = core::StateVectorArray<STATE_DIM, SCALAR>(K_ + 1, x0);
210  u_ = core::ControlVectorArray<CONTROL_DIM, SCALAR>(K_, u0);
211 
212  core::StateMatrix<STATE_DIM, SCALAR> A;
213  core::StateControlMatrix<STATE_DIM, CONTROL_DIM, SCALAR> B;
214  linearSystem.getAandB(x0, u0, 0, A, B);
215 
216  A_ = core::StateMatrixArray<STATE_DIM, SCALAR>(K_, A);
217  B_ = core::StateControlMatrixArray<STATE_DIM, CONTROL_DIM, SCALAR>(K_, B);
218  b_ = core::StateVectorArray<STATE_DIM, SCALAR>(K_ + 1, offset + x0 - A * x0 - B * u0);
219 
220  // feed current state and control to cost function
221  costFunction.setCurrentStateAndControl(x0, u0, 0);
222 
223  // intermediate stage
224  Q_ = core::StateMatrixArray<STATE_DIM, SCALAR>(K_ + 1, costFunction.stateSecondDerivativeIntermediate() * dt);
225  P_ =
226  core::FeedbackArray<STATE_DIM, CONTROL_DIM, SCALAR>(K_, costFunction.stateControlDerivativeIntermediate() * dt);
227  R_ = core::ControlMatrixArray<CONTROL_DIM, SCALAR>(K_, costFunction.controlSecondDerivativeIntermediate() * dt);
228 
229  qv_ = core::StateVectorArray<STATE_DIM, SCALAR>(
230  K_ + 1, costFunction.stateDerivativeIntermediate() * dt - Q_.front() * x0 - P_.front().transpose() * u0);
231  rv_ = core::ControlVectorArray<CONTROL_DIM, SCALAR>(
232  K_, costFunction.controlDerivativeIntermediate() * dt - R_.front() * u0 - P_.front() * x0);
233 
234  // final stage
235  Q_[K_] = costFunction.stateSecondDerivativeTerminal();
236  qv_[K_] = costFunction.stateDerivativeTerminal() - Q_[K_] * x0;
237 
238  hasBoxConstraints_ = false;
239  hasGenConstraints_ = false;
240 }
