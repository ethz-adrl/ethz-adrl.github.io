393 template <size_t STATE_DIM, size_t CONTROL_DIM, size_t P_DIM, size_t V_DIM, typename SCALAR, bool CONTINUOUS>
394 bool NLOCBackendBase<STATE_DIM, CONTROL_DIM, P_DIM, V_DIM, SCALAR, CONTINUOUS>::rolloutSingleShot(const size_t threadId,
395  const size_t k,
396  ControlVectorArray& u_local,
397  StateVectorArray& x_local,
398  const StateVectorArray& x_ref_lqr,
399  StateVectorArray& xShot,
400  StateSubsteps& substepsX,
401  ControlSubsteps& substepsU,
402  std::atomic_bool* terminationFlag) const
403 {
404  const int K_local = K_;
405 
406  if (u_local.size() < (size_t)K_)
407  throw std::runtime_error("rolloutSingleShot: u_local is too short.");
408  if (x_local.size() < (size_t)K_ + 1)
409  throw std::runtime_error("rolloutSingleShot: x_local is too short.");
410  if (xShot.size() < (size_t)K_ + 1)
411  throw std::runtime_error("rolloutSingleShot: xShot is too short.");
412 
413  xShot[k] = x_local[k]; // initialize
414 
416  int K_stop = k + getNumStepsPerShot();
417  if (K_stop > K_local)
418  K_stop = K_local;
419 
420  // for each control step
421  for (int i = (int)k; i < K_stop; i++)
422  {
423  if (terminationFlag && *terminationFlag)
424  return false;
425 
426  if (i > (int)k)
427  {
428  xShot[i] = xShot[i - 1];
429  }
430 
431  // Todo: the order here is not optimal. In some cases, we will overwrite x_ref_lqr immediately in the next step!
432  if (settings_.closedLoopShooting) // overwrite control
433  u_local[i] += L_[i] * (xShot[i] - x_ref_lqr[i]);
434 
436  if (i > (int)k)
437  {
438  x_local[i] = xShot[i - 1];
439  }
440 
441  state_vector_t stateNext;
442  systemInterface_->propagateControlledDynamics(xShot[i], i, u_local[i], stateNext, threadId);
443  xShot[i] = stateNext;
444 
445  if (i == K_local - 1)
446  x_local[K_local] = xShot[K_local - 1];
447 
448 
449  // check if nan
450  for (size_t j = 0; j < STATE_DIM; j++)
451  {
452  if (std::isnan(x_local[i](j)))
453  {
454  x_local.resize(K_local + 1,
455  ct::core::StateVector<STATE_DIM, SCALAR>::Constant(std::numeric_limits<SCALAR>::quiet_NaN()));
456  u_local.resize(K_local,
457  ct::core::ControlVector<CONTROL_DIM, SCALAR>::Constant(std::numeric_limits<SCALAR>::quiet_NaN()));
458  return false;
459  }
460  }
461  for (size_t j = 0; j < CONTROL_DIM; j++)
462  {
463  if (std::isnan(u_local[i](j)))
464  {
465  x_local.resize(K_local + 1,
466  ct::core::StateVector<STATE_DIM, SCALAR>::Constant(std::numeric_limits<SCALAR>::quiet_NaN()));
467  u_local.resize(K_local,
468  ct::core::ControlVector<CONTROL_DIM, SCALAR>::Constant(std::numeric_limits<SCALAR>::quiet_NaN()));
469  std::cout << "control unstable" << std::endl;
470  return false;
471  }
472  }
473 
474  // get substeps for later sensitvity calculation
475  systemInterface_->getSubstates(substepsX[i], threadId);
476  systemInterface_->getSubcontrols(substepsU[i], threadId);
477  }
478 
479  return true;
480 }

632 template <size_t STATE_DIM, size_t CONTROL_DIM, size_t P_DIM, size_t V_DIM, typename SCALAR, bool CONTINUOUS>
633 void NLOCBackendBase<STATE_DIM, CONTROL_DIM, P_DIM, V_DIM, SCALAR, CONTINUOUS>::executeLQApproximation(size_t threadId,
634  size_t k)
635 {
636  LQOCProblem_t& p = *lqocProblem_;
637  const scalar_t& dt = settings_.dt;
638 
639  assert(lqocProblem_ != nullptr);
640 
641  assert(&lqocProblem_->A_[k] != nullptr);
642  assert(&lqocProblem_->B_[k] != nullptr);
643  assert(&lqocProblem_->b_[k] != nullptr);
644 
645  assert(lqocProblem_->A_.size() > k);
646  assert(lqocProblem_->B_.size() > k);
647  assert(lqocProblem_->b_.size() > k);
648 
649 
651  // compute A_n and B_n
652  systemInterface_->setSubstepTrajectoryReference(substepsX_, substepsU_, threadId);
653  systemInterface_->getAandB(x_[k], u_ff_[k], xShot_[k], (int)k, settings_.K_sim, p.A_[k], p.B_[k], threadId);
654 
655  // compute dynamics offset term b_n
656  p.b_[k] = d_[k] + x_[k + 1] - p.A_[k] * x_[k] - p.B_[k] * u_ff_[k];
657 
658  // feed current state and control to cost function
659  costFunctions_[threadId]->setCurrentStateAndControl(x_[k], u_ff_[k], dt * k);
660 
661  // By using the following order of evaluations, we avoid caching matrices
662  // second derivative w.r.t state
663  p.Q_[k] = costFunctions_[threadId]->stateSecondDerivativeIntermediate() * dt;
664  // second derivative w.r.t control
665  p.R_[k] = costFunctions_[threadId]->controlSecondDerivativeIntermediate() * dt;
666  // cross terms
667  p.P_[k] = costFunctions_[threadId]->stateControlDerivativeIntermediate() * dt;
668 
669  // derivative of cost with respect to state
670  p.qv_[k] =
671  costFunctions_[threadId]->stateDerivativeIntermediate() * dt - p.Q_[k] * x_[k] - p.P_[k].transpose() * u_ff_[k];
672  // derivative of cost with respect to control
673  p.rv_[k] = costFunctions_[threadId]->controlDerivativeIntermediate() * dt - p.R_[k] * u_ff_[k] - p.P_[k] * x_[k];
674 
675  // p.q_[k] = ... // not evaluated since we don't need it in GNMS/iLQR -- WARNING, potentially implement when using a different QP solver
676 
677  // set current reference trajectories x_n and u_n
678  p.x_[k] = x_[k];
679  p.u_[k] = u_ff_[k];
680 }

1000

1003  }

1012  resetDefects();

1016  {

1019  std::cout << "[LineSearch]: Defect norm last rollout:\t" << d_norm_ << std::endl;
1020  std::cout << "[LineSearch]: err box constr last rollout:\t" << e_box_norm_ << std::endl;
1021  std::cout << "[LineSearch]: err gen constr last rollout:\t" << e_gen_norm_ << std::endl;
1022  std::cout << "[LineSearch]: Merit of last rollout:\t" << lowestCost_ << std::endl;
1023  }
1024 
1025  alphaBest_ = performLineSearch();
1026 
1027  if (settings_.lineSearchSettings.debugPrint)
1028  {

1081  ControlSubsteps& substepsU,

1088  e_gen_norm = 0.0;
1089 
1090  if (terminationFlag && *terminationFlag)
1091  return;
1092 
1093  // update feedforward with weighting alpha
1094  u_alpha = u_ff_new * alpha + u_ff_prev_ * (1 - alpha);
1095 
1096  // update state decision variables with weighting alpha
1097  x_alpha = x_new * alpha + x_prev_ * (1 - alpha);
1098 
1099 
1100  if (terminationFlag && *terminationFlag)
1101  return;
1102

1107

1109  return;

1113  {
1115  defectNorm = computeDefectsNorm<1>(defects_recorded);
1116 
1117  if (terminationFlag && *terminationFlag)
1118  return;
1119 
1120  computeCostsOfTrajectory(threadId, x_alpha, u_alpha, intermediateCost, finalCost);
1121 
1122  if (terminationFlag && *terminationFlag)
1123  return;
1124

1126  if (boxConstraints_[threadId] != nullptr)
1127  computeBoxConstraintErrorOfTrajectory(threadId, x_alpha, u_alpha, e_box_norm);
1128 
1129  if (terminationFlag && *terminationFlag)
1130  return;
1131

1133  computeGeneralConstraintErrorOfTrajectory(threadId, x_alpha, u_alpha, e_gen_norm);
1134  }
1135  else
1136  {
1137  if (settings_.debugPrint)
1138  {
