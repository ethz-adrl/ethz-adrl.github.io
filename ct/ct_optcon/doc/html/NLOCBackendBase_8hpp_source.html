611 template <typename ARRAY_TYPE, size_t ORDER>

625 template <size_t STATE_DIM, size_t CONTROL_DIM, size_t P_DIM, size_t V_DIM, typename SCALAR, bool CONTINUOUS>
626 template <typename ARRAY_TYPE, size_t ORDER>
627 SCALAR NLOCBackendBase<STATE_DIM, CONTROL_DIM, P_DIM, V_DIM, SCALAR, CONTINUOUS>::computeDiscreteArrayNorm(
628  const ARRAY_TYPE& a,
629  const ARRAY_TYPE& b) const
630 {
631  assert(a.size() == b.size());
632 
633  SCALAR norm = 0.0;
634 
635  for (size_t k = 0; k < a.size(); k++)
636  {
637  norm += (a[k] - b[k]).template lpNorm<ORDER>();
638  }
639  return norm;
640 }
