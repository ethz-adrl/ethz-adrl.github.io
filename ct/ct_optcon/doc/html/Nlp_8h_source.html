136  void evaluateHessian(const int nele_hes, MapVecXs& hes, const SCALAR obj_fac, MapConstVecXs& lambda)
137  {
138 #if EIGEN_VERSION_AT_LEAST(3, 3, 0)
139 
140  // store objective-factor in matrix to ensure compatibility with other derivative-classes
141  Eigen::Matrix<double, 1, 1> omega;
142  omega << obj_fac;
143 
144  // evaluate Hessian values
145  Eigen::VectorXd hessianCostValues, hessianConstraintsValues;
146 
147  if (costCodegen_)
148  hessianCostValues = costCodegen_->sparseHessianValues(optVariables_->getOptimizationVars(), omega);
149  else
150  costEvaluator_->sparseHessianValues(optVariables_->getOptimizationVars(), omega, hessianCostValues);
151 
152 
153  if (constraintsCodegen_)
154  hessianConstraintsValues =
155  constraintsCodegen_->sparseHessianValues(optVariables_->getOptimizationVars(), lambda);
156  else
157  hessianConstraintsValues = constraints_->sparseHessianValues(optVariables_->getOptimizationVars(), lambda);
158 
159 
160  // collect all Hessian values (from both constraints and cost) in one vector of triplets
161  // here, we re-use the the sparsity information collected in getNonZeroHessianCount()
162  std::vector<Eigen::Triplet<SCALAR>, Eigen::aligned_allocator<Eigen::Triplet<SCALAR>>> triplets;
163  triplets.reserve(iRowHessianCost_.rows() + iRowHessianConstraints_.rows());
164 
165  for (int i = 0; i < iRowHessianCost_.rows(); i++)
166  {
167  triplets.push_back(Eigen::Triplet<SCALAR>(iRowHessianCost_(i), jColHessianCost_(i), hessianCostValues(i)));
168  }
169  for (int i = 0; i < iRowHessianConstraints_.rows(); i++)
170  {
171  triplets.push_back(Eigen::Triplet<SCALAR>(
172  iRowHessianConstraints_(i), jColHessianConstraints_(i), hessianConstraintsValues(i)));
173  }
174 
175  // setting the sparse Hessian from Eigen-triplets automatically resolves (e.g. adds) entries at same indices
176  Hessian_eval_.setFromTriplets(triplets.begin(), triplets.end());
177 
178  // IPOPT and SNOPT only expect a triangular Hessian matrix (symmetry)
179  // enforce triangular view
180  // todo: is there a nicer in-place conversion to triangularView?
181  Hessian_eval_ = Hessian_eval_.template triangularView<Eigen::Lower>();
182 
183  hes = Eigen::Map<Eigen::VectorXd>(Hessian_eval_.valuePtr(), nele_hes, 1);
184 
185 #else
186  throw std::runtime_error(
187  "Nlp::evaluateHessian() only available for Eigen 3.3 and newer. Please change solver settings to NOT use "
188  "exact Hessians");
189 #endif
190  }

275  size_t getNonZeroHessianCount()
276  {
277 #if EIGEN_VERSION_AT_LEAST(3, 3, 0)
278 
279  // the sparse Eigen-matrices need to be resized properly, which happens in this step.
280  // todo: need to assert that getNonZeroHessianCount() gets called before the first call to a Hessian evaluation.
281  Hessian_eval_.resize(optVariables_->size(), optVariables_->size());
282  Hessian_sparsity_.resize(optVariables_->size(), optVariables_->size());
283 
284  iRowHessianCost_.setZero();
285  jColHessianCost_.setZero();
286  iRowHessianConstraints_.setZero();
287  jColHessianConstraints_.setZero();
288 
289 
290  // evaluate sparsity patterns
291  if (constraintsCodegen_)
292  constraintsCodegen_->getSparsityPatternHessian(iRowHessianConstraints_, jColHessianConstraints_);
293  else
294  constraints_->getSparsityPatternHessian(
295  iRowHessianConstraints_, jColHessianConstraints_, optVariables_->size());
296 
297  if (costCodegen_)
298  costCodegen_->getSparsityPatternHessian(iRowHessianCost_, jColHessianCost_);
299  else
300  costEvaluator_->getSparsityPatternHessian(iRowHessianCost_, jColHessianCost_);
301 
302 
303  // collect all Hessian sparsity values (from both constraints and cost) in one vector of triplets
304  std::vector<Eigen::Triplet<SCALAR>, Eigen::aligned_allocator<Eigen::Triplet<SCALAR>>> triplets;
305  triplets.reserve(iRowHessianCost_.rows() + iRowHessianConstraints_.rows());
306 
307  for (int i = 0; i < iRowHessianCost_.rows(); ++i)
308  triplets.push_back(Eigen::Triplet<SCALAR>(iRowHessianCost_(i), jColHessianCost_(i), SCALAR(1.0)));
309 
310  for (int i = 0; i < iRowHessianConstraints_.rows(); ++i)
311  triplets.push_back(
312  Eigen::Triplet<SCALAR>(iRowHessianConstraints_(i), jColHessianConstraints_(i), SCALAR(1.0)));
313 
314  // setting the sparse Hessian from Eigen-triplets automatically resolves (e.g. adds) entries at same indices
315  Hessian_sparsity_.setFromTriplets(triplets.begin(), triplets.end());
316 
317  // IPOPT and SNOPT only expect a triangular Hessian matrix (symmetry)
318  // enforce triangular view
319  // todo: is there a nicer in-place conversion to triangularView?
320  Hessian_sparsity_ = Hessian_sparsity_.template triangularView<Eigen::Lower>();
321 
322  // lastly, collect and combine the sparsity as filled into the helper-matrix and store in Eigen::Vectors.
323  std::vector<int> iRowHessianStdVec;
324  std::vector<int> jColHessianStdVec;
325  for (int k = 0; k < Hessian_sparsity_.outerSize(); ++k)
326  {
327  for (typename Eigen::SparseMatrix<SCALAR>::InnerIterator it(Hessian_sparsity_, k); it; ++it)
328  {
329  iRowHessianStdVec.push_back(it.row());
330  jColHessianStdVec.push_back(it.col());
331  }
332  }
333  iRowHessian_ = Eigen::Map<Eigen::VectorXi>(iRowHessianStdVec.data(), iRowHessianStdVec.size(), 1);
334  jColHessian_ = Eigen::Map<Eigen::VectorXi>(jColHessianStdVec.data(), jColHessianStdVec.size(), 1);
335 
336  // the number of non-zero elements is equal to the number rows
337  size_t nonZerosHessian = iRowHessian_.rows();
338  return nonZerosHessian;
339 #else
340  throw std::runtime_error(
341  "Nlp::getNonZeroHessianCount() only available for Eigen 3.3 and newer. Please change solver settings to "
342  "NOT use "
343  "exact Hessians or upgrade Eigen version.");
344  return 0;
345 #endif
346  }
