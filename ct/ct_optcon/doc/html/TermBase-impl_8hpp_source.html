145 template <size_t STATE_DIM, size_t CONTROL_DIM, typename SCALAR_EVAL, typename SCALAR>
146 void TermBase<STATE_DIM, CONTROL_DIM, SCALAR_EVAL, SCALAR>::loadTimeActivation(const std::string& filename,
147  const std::string& termName,
148  bool verbose)
149 {
150  if (verbose)
151  std::cout << "TermBase: loading TimeActivation ..." << std::endl;
152 
153  boost::property_tree::ptree pt;
154  boost::property_tree::read_info(filename, pt);
155 
156  try
157  {
158  std::string activationKind = pt.get<std::string>(termName + ".time_activation" + ".kind");
159  boost::algorithm::to_lower(activationKind);
160  std::shared_ptr<ct::core::tpl::ActivationBase<SCALAR_EVAL>> c_i;
161  CT_LOADABLE_ACTIVATIONS(SCALAR_EVAL);
162  c_i->loadConfigFile(filename, termName + ".time_activation", verbose);
163  if (!c_i)
164  {
165  throw std::runtime_error("Activation type \"" + activationKind + "\" not supported");
166  }
167  else
168  {
169  c_i_ = c_i;
170  if (verbose)
171  c_i_->printInfo();
172  }
173  } catch (std::exception& e)
174  {
175  if (verbose)
176  std::cout << "TermBase: encountered exception while loading TimeActivation." << std::endl;
177  return;
178  }
179 }
