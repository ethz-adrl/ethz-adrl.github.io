104  const Eigen::Matrix<SC, CONTROL_DIM, 1>& u,

106 {

110  return (xDiff.transpose() * Q_.template cast<SC>() * xDiff)(0, 0) *
111  (uDiff.transpose() * R_.template cast<SC>() * uDiff)(0, 0);
112 }
