51  void static Crout(const Eigen::Matrix<T, rowAndCol, rowAndCol>& S, Eigen::Matrix<T, rowAndCol, rowAndCol>& D)
52  {
53  for (int k = 0; k < S.rows(); ++k)
54  {
55  for (int i = k; i < S.rows(); ++i)
56  {
57  T sum(0.);
58  for (int p = 0; p < k; ++p)
59  sum += D(i, p) * D(p, k);
60  D(i, k) = S(i, k) - sum; // not dividing by diagonals
61  }
62  for (int j = k + 1; j < S.rows(); ++j)

75  const Eigen::Matrix<T, rowAndCol, 1>& b,
