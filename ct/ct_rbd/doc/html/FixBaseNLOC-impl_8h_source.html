112 template <class FIX_BASE_FD_SYSTEM>
113 void FixBaseNLOC<FIX_BASE_FD_SYSTEM>::initializeDirectInterpolation(const RobotState_t& x0,
114  const RobotState_t& xf,
115  const core::Time& tf,
116  const int N,
117  ct::core::ControlVectorArray<NJOINTS, SCALAR>& uff_array,
118  ct::core::StateVectorArray<STATE_DIM, SCALAR>& x_array,
119  FeedbackMatrix K)
120 {
121  uff_array.resize(N);
122  x_array.resize(N + 1);
123 
124  // temporary variables
125  ControlVector uff_torque;
126 
127  for (int i = 0; i < N + 1; i++)
128  {
129  RobotState_t state_temp = x0;
130  state_temp.joints().toImplementation() =
131  x0.joints().toImplementation() +
132  (xf.joints().toImplementation() - x0.joints().toImplementation()) * SCALAR(i) / SCALAR(N);
133 
134  if (i < N)
135  {
136  uff_torque = system_->computeIDTorques(state_temp.joints());
137  if (ACTUATOR_STATE_DIM > 0) // if there are actuator dynamics
138  {
139  uff_array[i].setZero(); // todo compute this from act.state
140  }
141  else // actuator dynamics off
142  {
143  uff_array[i] = uff_torque;
144  }
145  }
146 
147  if (i > 0 && ACTUATOR_STATE_DIM > 0)
148  {
149  // direct interpolation for actuator state may not make sense. Improve using actuator model
150  state_temp.actuatorState() =
151  system_->getActuatorDynamics()->computeStateFromOutput(state_temp.joints(), uff_torque);
152  }
153 
154  x_array[i] = state_temp.toStateVector();
155  }
156 
157  FeedbackArray u0_fb(N, K);
158 
159  typename NLOptConSolver::Policy_t policy(x_array, uff_array, u0_fb, getSettings().dt);
160 
161  std::cout << "init with " << x0.toStateVector() << std::endl;
162 
163  nlocSolver_->changeInitialState(x0.toStateVector());
164  nlocSolver_->changeTimeHorizon(tf);
165  nlocSolver_->setInitialGuess(policy);
166 }
