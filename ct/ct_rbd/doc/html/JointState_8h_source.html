91  template <typename T>
92  void toUniquePosition(T lowerLimitVec, double tolerance = 0.0)
93  {
94  assert(lowerLimitVec.size() == NJOINTS && "Wrong limit dimensions");
95  for (size_t i = 0; i < NJOINTS; ++i)
96  {
97  // compute the integer of multiple of 2*PI to substract
98  int k_lower = std::floor((getPosition(i) - (lowerLimitVec[i] - tolerance)) / (2 * M_PI));
99  int k_upper = std::floor((getPosition(i) - (lowerLimitVec[i] + tolerance)) / (2 * M_PI));
100 
101  if (abs(k_lower) <= abs(k_upper))
102  {
103  if (k_lower != 0)
104  getPosition(i) -= k_lower * 2 * M_PI;
105  }
106  else if (k_upper != 0)
107  getPosition(i) -= k_upper * 2 * M_PI;
108  }
109  }
