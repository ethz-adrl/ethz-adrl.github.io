170  void fromStateVectorQuaternion(const state_vector_quat_t& state)
171  {
172  try
173  {
174  base().pose().setFromRotationQuaternion(kindr::RotationQuaternion<SCALAR>(state.template head<4>()));
175  base().pose().position().toImplementation() = state.template segment<3>(4);
176  joints().getPositions() = state.template segment<NJOINTS>(7);
177  base().velocities().getRotationalVelocity().toImplementation() = state.template segment<3>(7 + NJOINTS);
178  base().velocities().getTranslationalVelocity().toImplementation() = state.template segment<3>(10 + NJOINTS);
179  joints().getVelocities() = state.template tail<NJOINTS>();
180  } catch (std::exception& e)
181  {
182  std::cout << "Conversion from State Vector to KindrTypes failed." << std::endl;
183  std::cout << "Error: " << e.what() << std::endl;
184  std::cout << "State is: " << state.transpose() << std::endl;
185  throw std::runtime_error("Conversion from State Vector to KindrTypes failed.");
186  }
187  }
