148  kindr::RotationMatrix<SCALAR> getRotationMatrix() const
149  {
150  if (storedAsEuler())

154  Eigen::AngleAxis<SCALAR> yawAngle(euler_.toImplementation()(2), Eigen::Matrix<SCALAR, 3, 1>::UnitZ());

254  RigidBodyPose<SCALAR> inReferenceFrame(const RigidBodyPose<SCALAR>& ref_frame) const
255  {

282  Vector3s rotateBaseToInertia(const Vector3s& vector) const
283  {
284  Vector3s out;
285 
286  if (storedAsEuler())
287  {
288  kindr::RotationMatrix<SCALAR> rotationMatrix = getRotationMatrix();
289 
290  // incredibly ugly hack to get along with different types
291  Eigen::Matrix<SCALAR, 3, 1> vec_temp;
292  vec_temp << vector(0), vector(1), vector(2);
293 
294  Eigen::Matrix<SCALAR, 3, 1> result = rotationMatrix.toImplementation() * vec_temp;
295  out = (Vector3s)result;
296 
297  // return euler_.rotate(vector); // temporarily replaced -- the kindr rotate() method is not auto-diffable
298  }
299  else
300  {
301  out = quat_.rotate(vector);
302  }
303 
304  return out;
305  }
