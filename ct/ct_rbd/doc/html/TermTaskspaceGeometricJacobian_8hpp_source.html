127  virtual double evaluate(const Eigen::Matrix<double, STATE_DIM, 1>& x,
128  const Eigen::Matrix<double, CONTROL_DIM, 1>& u,
129  const double& t) override
130  {
131  // transform the robot state vector into a CT RBDState
132  RBDState rbdState;
133  rbdState.jointPositions() = x.template head<KINEMATICS::NJOINTS>();
134 
135  // position difference in world frame
136  Eigen::Matrix<double, 3, 1> xCurr =
137  kinematics_.getEEPositionInWorld(eeInd_, rbdState.basePose(), rbdState.jointPositions()).toImplementation();
138  Eigen::Matrix<double, 3, 1> xDiff = xCurr - x_ref_;
139 
140  // compute the cost from the position error
141  double pos_cost = (xDiff.transpose() * Q_pos_ * xDiff)(0, 0);
142 
143  // get current end-effector rotation in world frame
144  Eigen::Matrix<double, 3, 3> ee_rot = kinematics_.getEERotInBase(eeInd_, rbdState.jointPositions());
145 
146  // compute a measure for the difference between current rotation and desired rotation and compute cost based on the orientation error
147  // for the intuition behind, consider the following posts:
148  // https://math.stackexchange.com/a/87698
149  // https://math.stackexchange.com/a/773635
150  // Eigen::Matrix<double, 3, 3> ee_R_diff = w_R_ref.transpose() * ee_rot;
151 
152  // compute rotation penalty using the squared Frobenius norm of (R_diff-I)
153  // double rot_cost = Q_rot_ * (ee_R_diff - Eigen::Matrix<double, 3, 3>::Identity()).squaredNorm();
154 
155  return pos_cost; // + rot_cost;
156  }

219  void loadConfigFile(const std::string& filename, const std::string& termName, bool verbose = false) override
220  {
221  if (verbose)
222  std::cout << "Loading TermTaskspaceGeometricJacobian from file " << filename << std::endl;
223 
224  ct::optcon::loadScalarCF(filename, "eeId", eeInd_, termName);
225  ct::optcon::loadScalarCF(filename, "Q_rot", Q_rot_, termName);
226  ct::optcon::loadMatrixCF(filename, "Q_pos", Q_pos_, termName);
227 
228  Eigen::Matrix<double, 3, 1> w_p_ref;
229  ct::optcon::loadMatrixCF(filename, "x_des", w_p_ref, termName);
230  setReferencePosition(w_p_ref);
231 
232 
233  // try loading euler angles
234  if (verbose)
235  std::cout << "trying to load Euler angles" << std::endl;
236  try
237  {
238  Eigen::Vector3d eulerXyz;
239  ct::optcon::loadMatrixCF(filename, "eulerXyz_des", eulerXyz, termName);
240  Eigen::Quaternion<double> quat_des(Eigen::AngleAxisd(eulerXyz(0), Eigen::Vector3d::UnitX()) *
241  Eigen::AngleAxisd(eulerXyz(1), Eigen::Vector3d::UnitY()) *
242  Eigen::AngleAxisd(eulerXyz(2), Eigen::Vector3d::UnitZ()));
243  setReferenceOrientation(quat_des);
244  if (verbose)
245  std::cout << "Read desired Euler Angles Xyz as = \n" << eulerXyz.transpose() << std::endl;
246  } catch (const std::exception& e)
247  {
248  throw std::runtime_error(
249  "Failed to load TermTaskspaceGeometricJacobian, could not find a desired end effector orientation in "
250  "file.");
251  }
252 
253  if (verbose)
254  {
255  std::cout << "Read eeId as eeId = \n" << eeInd_ << std::endl;
256  std::cout << "Read Q_pos as Q_pos = \n" << Q_pos_ << std::endl;
257  std::cout << "Read Q_rot as Q_rot = \n" << Q_rot_ << std::endl;
258  std::cout << "Read x_des as x_des = \n" << getReferencePosition().transpose() << std::endl;
259  }

261  // new initialization required
262  setup();
