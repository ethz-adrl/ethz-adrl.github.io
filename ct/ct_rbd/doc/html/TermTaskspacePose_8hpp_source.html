228  template <typename SC>
229  SC evalLocal(const Eigen::Matrix<SC, STATE_DIM, 1>& x, const Eigen::Matrix<SC, CONTROL_DIM, 1>& u, const SC& t)
230  {
231  using Matrix3Tpl = Eigen::Matrix<SC, 3, 3>;
232 
233  // transform the robot state vector into a CT RBDState
234  RBDState<KINEMATICS::NJOINTS, SC> rbdState = setStateFromVector(x);
235 
236  // position difference in world frame
237  Eigen::Matrix<SC, 3, 1> xDiff =
238  kinematics_.getEEPositionInWorld(eeInd_, rbdState.basePose(), rbdState.jointPositions())
239  .toImplementation() -
240  w_p_ref_.template cast<SC>();
241 
242  // compute the cost based on the position error
243  SC pos_cost = (xDiff.transpose() * Q_pos_.template cast<SC>() * xDiff)(0, 0);
244 
245 
246  // get current end-effector rotation in world frame
247  Matrix3Tpl ee_rot = kinematics_.getEERotInWorld(eeInd_, rbdState.basePose(), rbdState.jointPositions());
248 
249  // compute a measure for the difference between current rotation and desired rotation and compute cost based on the orientation error
250  // for the intuition behind, consider the following posts:
251  // https://math.stackexchange.com/a/87698
252  // https://math.stackexchange.com/a/773635
253  Matrix3Tpl ee_rot_diff = w_R_ref_.template cast<SC>().transpose() * ee_rot;
254 
255  SC rot_cost =
256  (SC)Q_rot_ * (ee_rot_diff - Matrix3Tpl::Identity()).squaredNorm(); // the frobenius norm of (R_diff-I)
257 
258  return pos_cost + rot_cost;
259  }
