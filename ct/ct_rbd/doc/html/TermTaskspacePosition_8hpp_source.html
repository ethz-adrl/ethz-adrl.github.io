115  {
116  RBDState<KINEMATICS::NJOINTS, SC> rbdState = setStateFromVector<FB>(x);
117 
118  Eigen::Matrix<SC, 3, 1> xDiff =
119  kinematics_.getEEPositionInWorld(eeInd_, rbdState.basePose(), rbdState.jointPositions())
120  .toImplementation() -
121  pos_ref_.template cast<SC>();

123  SC cost = (xDiff.transpose() * QTaskSpace_.template cast<SC>() * xDiff)(0, 0);
